void
monitorHabitat ()
{

  string habitat;
  fstream file;
  file.open ("habitats.txt");

  while (!file.eof ())
    {
      file >> habitat;
      cout << habitat << endl;
    }
  file.close ();
}


void

  fstream file;

}
